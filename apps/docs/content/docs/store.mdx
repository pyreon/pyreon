---
title: "Store"
description: "defineStore, resetStore, and resetAllStores for global state management."
---

`@pyreon/store` provides composable global state through `defineStore`. Stores are lazily
initialized singletons — the setup function runs once the first time the store is used.

<Callout type="info" title="Stores are singletons">
  Every call to a store function returns the same instance. The setup function runs
  exactly once, lazily, on first use. This means stores are safe to import anywhere
  without worrying about duplicate state.
</Callout>

## Installation

```bash title="Terminal"
bun add @pyreon/store
```

## Defining a Store

<Steps>
<Step>

### Define the store

```ts title="src/stores/counter.ts"
import { defineStore } from "@pyreon/store"
import { signal, computed } from "@pyreon/reactivity"

export const useCounterStore = defineStore(() => {
  const count = signal(0)
  const doubled = computed(() => count() * 2)

  function increment() {
    count.update(n => n + 1)
  }

  function reset() {
    count.set(0)
  }

  return { count, doubled, increment, reset }
})
```

</Step>
<Step>

### Use it in any component

```tsx title="src/Counter.tsx"
import { useCounterStore } from "./stores/counter"

function Counter() {
  const store = useCounterStore()

  return (
    <div>
      <p>Count: {() => store.count()}</p>
      <p>Doubled: {() => store.doubled()}</p>
      <button onClick={store.increment}>+1</button>
      <button onClick={store.reset}>Reset</button>
    </div>
  )
}
```

</Step>
<Step>

### Reset if needed

```ts
import { resetStore, resetAllStores } from "@pyreon/store"
import { useCounterStore } from "./stores/counter"

resetStore(useCounterStore)    // re-runs useCounterStore's setup on next call
resetAllStores()               // resets every registered store (useful in tests)
```

</Step>
</Steps>

## Store with Async Actions

```ts title="src/stores/user.ts"
import { defineStore } from "@pyreon/store"
import { signal } from "@pyreon/reactivity"

interface User { id: number; name: string }

export const useUserStore = defineStore(() => {
  const user = signal<User | null>(null)
  const loading = signal(false)
  const error = signal<string | null>(null)

  async function fetchUser(id: number) {
    loading.set(true)
    error.set(null)
    try {
      const res = await fetch(`/api/users/${id}`)
      user.set(await res.json())
    } catch (e) {
      error.set(String(e))
    } finally {
      loading.set(false)
    }
  }

  return { user, loading, error, fetchUser }
})
```
