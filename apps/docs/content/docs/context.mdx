---
title: "Context"
description: "createContext, useContext, and withContext for dependency injection in component trees."
---

Nova's context system provides dependency injection for component trees. It lets ancestor
components provide values that any descendant can read without passing props through every
intermediate layer.

## The Pattern

<Steps>
<Step>

### Create a context

```ts title="src/ThemeContext.ts"
import { createContext } from "@nova/core"

export const ThemeContext = createContext<"light" | "dark">("light")
//                                                           ↑ default value
```

</Step>
<Step>

### Provide a value

<Tabs items={["JSX (ThemeContext.Provider)", "withContext"]}>
<Tab value="JSX (ThemeContext.Provider)">
```tsx title="src/App.tsx"
import { ThemeContext } from "./ThemeContext"

function App() {
  return (
    <ThemeContext.Provider value="dark">
      <Sidebar />
      <Main />
    </ThemeContext.Provider>
  )
}
```
</Tab>
<Tab value="withContext">
```tsx title="src/App.tsx"
import { withContext } from "@nova/core"
import { ThemeContext } from "./ThemeContext"

function App() {
  return withContext(ThemeContext, "dark", () => (
    <>
      <Sidebar />
      <Main />
    </>
  ))
}
```
</Tab>
</Tabs>

</Step>
<Step>

### Consume the value

```tsx title="src/Button.tsx"
import { useContext } from "@nova/core"
import { ThemeContext } from "./ThemeContext"

function Button({ label }: { label: string }) {
  const theme = useContext(ThemeContext)  // "dark"

  return (
    <button class={`btn btn-${theme}`}>{label}</button>
  )
}
```

</Step>
</Steps>

## Reactive Context Values

If you provide a signal as the context value, consumers can read it reactively.

```tsx title="src/App.tsx"
import { signal } from "@nova/reactivity"
import { createContext } from "@nova/core"

const ThemeContext = createContext(signal<"light" | "dark">("light"))

function App() {
  const theme = signal<"light" | "dark">("light")

  return (
    <ThemeContext.Provider value={theme}>
      <button onClick={() => theme.update(t => t === "light" ? "dark" : "light")}>
        Toggle
      </button>
      <ThemedPage />
    </ThemeContext.Provider>
  )
}

function ThemedPage() {
  const theme = useContext(ThemeContext)
  return <div class={() => `page page-${theme()}`}>...</div>
}
```

## API Reference

<TypeTable
  type={{
    "createContext(defaultValue)": {
      description: "Creates a context with a default value used when no Provider is found above.",
      type: "Context<T>",
    },
    "useContext(context)": {
      description: "Reads the nearest provided value for the context.",
      type: "T",
    },
    "withContext(context, value, fn)": {
      description: "Runs fn with the context set to value. Returns fn's result.",
      type: "ReturnType<fn>",
    },
    "Context.Provider": {
      description: "JSX component that provides a value to all descendants.",
      type: "ComponentFn<{ value: T; children?: VNodeChild }>",
    },
  }}
/>

<Callout type="warn" title="Context is captured at component setup time">
  `useContext` reads the context value that was active when the component function ran.
  It does not reactively update if the provider changes its value — unless the value itself
  is a signal.
</Callout>
