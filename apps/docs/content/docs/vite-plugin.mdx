---
title: "Vite Plugin"
description: "Setting up @pyreon/vite-plugin for JSX transforms and .pyreon single-file components."
---

`@pyreon/vite-plugin` configures Vite for Pyreon projects. It sets up the JSX transform, handles
`.pyreon` single-file components, and enables HMR.

## Installation

```bash title="Terminal"
bun add --dev @pyreon/vite-plugin
```

## Setup

<Steps>
<Step>

### Configure Vite

```ts title="vite.config.ts"
import { defineConfig } from "vite"
import pyreon from "@pyreon/vite-plugin"

export default defineConfig({
  plugins: [pyreon()],
})
```

</Step>
<Step>

### Update tsconfig.json

```json title="tsconfig.json"
{
  "compilerOptions": {
    "jsx": "react-jsx",
    "jsxImportSource": "@pyreon/core"
  }
}
```

</Step>
<Step>

### Write your first component

<Tabs items={[".tsx file", ".pyreon SFC"]}>
<Tab value=".tsx file">
```tsx title="src/Hello.tsx"
import { signal } from "@pyreon/reactivity"

export function Hello({ name }: { name: string }) {
  const clicked = signal(false)

  return (
    <div>
      <h1>Hello, {name}!</h1>
      <button onClick={() => clicked.set(true)}>
        {() => clicked() ? "Clicked!" : "Click me"}
      </button>
    </div>
  )
}
```
</Tab>
<Tab value=".pyreon SFC">
```html title="src/Hello.pyreon"
<template>
  <div>
    <h1>Hello, {{ name }}!</h1>
    <button @click="clicked = !clicked">
      {{ clicked ? "Clicked!" : "Click me" }}
    </button>
  </div>
</template>

<script>
export default {
  props: { name: String },
  setup(props) {
    const clicked = signal(false)
    return { clicked }
  }
}
</script>

<style scoped>
h1 { color: indigo; }
</style>
```
</Tab>
</Tabs>

</Step>
<Step>

### Mount and run

```ts title="src/main.ts"
import { mount } from "@pyreon/runtime-dom"
import App from "./App"

mount(<App />, document.getElementById("app")!)
```

```bash title="Terminal"
bun run dev
```

</Step>
</Steps>

## .pyreon Single-File Component Structure

<Files>
  <Folder name="src" defaultOpen>

      <File name="Button.pyreon" />
      <File name="Header.tsx" />
    </Folder>
    <File name="main.ts" />
  </Folder>
  <File name="vite.config.ts" />
  <File name="tsconfig.json" />
  <File name="index.html" />
</Files>

The `.pyreon` format compiles to a standard ES module — you can mix `.pyreon` and `.tsx` files
freely in the same project.
